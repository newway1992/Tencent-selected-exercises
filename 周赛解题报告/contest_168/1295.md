水题，给你一个整数数组 nums，请你返回其中位数为 偶数 的数字的个数

```
func findNumbers(nums []int) int {
	ans := 0
	for i := 0; i < len(nums); i++{
		count := 0
		for nums[i] != 0{
			count++
			nums[i] /= 10
		}
		if count & 1 == 0{
			ans++
		}
	}
	return ans
}
```
