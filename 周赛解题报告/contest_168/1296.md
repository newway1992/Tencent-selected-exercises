水题，给你一个整数数组 nums，请你返回其中位数为 偶数 的数字的个数

```
func isPossibleDivide(nums []int, k int) bool {
	ans := true
	count := len(nums)
	if count%k != 0 {
		ans = false
	}
	tally := make(map[int]int)
	for i := 0; i < count; i++ {
		tally[nums[i]]++
	}
	keys := make([]int, 0)
	for k, _ := range tally {
		keys = append(keys, k)
	}
	sort.Ints(keys)
	keyCount := len(keys)
	for i := 0; i < keyCount; i++{
		cnt := tally[keys[i]]
		if cnt > 0{
			for j := keys[i]; j < keys[i] + k;j++{
				if tally[j] >= cnt{
					tally[j] -= cnt
				}else{
					ans = false
					break
				}
			}
		}
		if !ans{
			break
		}
	}
	return ans
}
```
